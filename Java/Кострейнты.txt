UNIQUE 			//Значение какого-то поля из таблицы не должно повторяться
Пример:			 name VARCHAR(128) UNIQUE NOT NULL

CHECK ()		//Позволяет задать какое-то условие для конкретного поля при создание таблица.
Пример:			  date DATE  NOT NULL CHECK ( date >'01-01-2000' AND date < '01-01-2020' ) <-- В таблице не могут быть записи, меньше 2000 года и больше 2020

PRIMARY KEY		//Первичный ключ
FOREIGN KEY 		//Внешний ключ(свзять между тоблицами )
