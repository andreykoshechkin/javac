INNER JOIN 	//Отсекает записи из всех таблица у которых нет внешнего ключа. Если в таблице есть запись которая не ссылается на запись из другой таблице, то выборка ее не покажет
LEFT  JOIN	//Возвращает записи из таблицы слева, даже те у которых нет внешнего ключа
RIGHT JOIN	//Возвращает записи из таблицы справа, даже те у которых нет внешнего ключа
CROSS JOIN	//Вернет декартовое множество таблиц (все со всеми)
FULL  JOIN	//Вернет записи и те которых нет внешнего ключа и те у которых есть внешний ключ

---Не смотря на то, что в примере ниже, у компании Beeline нет сотрудников
-- мы все равно получаем 1 запись. Это связано с тем, что символ * в count
---в случае LEFT возвращает одну запись, поэтому лучше использовать count(eml.id)

SELECT
    com.name,
    com.date,
    count(eml.id)
FROM company com
LEFT JOIN employee eml ON eml.company_id = com.id
GROUP BY  com.id



---В примере ниже, мы отобразим все компании но значение в столбце username будет null
---Это связано с тем, что мы поставили условие AND, а это значит что мы ставим условие
---на еще не сгруппировавшийся значения, если мы будет использовать WHERE, то в таком случае
---where будет выполнено после групировки значений

SELECT com.name,
       eml.username
FROM company com
         LEFT JOIN employee eml ON com.id = eml.company_id
    WHERE com.name = 'Beeline'
