INNER JOIN 	//Отсекает записи из всех таблица у которых нет внешнего ключа. Если в таблице есть запись которая не ссылается на запись из другой таблице, то выборка ее не покажет
LEFT  JOIN	//Возвращает записи из таблицы слева, даже те у которых нет внешнего ключа
RIGHT JOIN	//Возвращает записи из таблицы справа, даже те у которых нет внешнего ключа
CROSS JOIN	//Вернет декартовое множество табли (все со всеми)
FULL  JOIN	//Вернет записи и те которых нет внешнего ключа и те у которых есть внешний ключ



SELECT *
FROM company	<<<< company это таблица слева
        LEFT JOIN employee e ON e.company_id = company.id

SELECT *
FROM company 
        RIGHT JOIN employee e ON e.company_id = company.id <<<< employee  это таблица справа
