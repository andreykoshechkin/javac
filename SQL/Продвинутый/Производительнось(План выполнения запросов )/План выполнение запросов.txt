1:	//Для того чтобы посмотрет план выполненния запроса, т.e посмотреть то как СУБД выполнила запрос, нужно использовать ключеваое слово explain
	EXPLAIN SELECT * FROM ticket;
	Seq Scan on ticket  (cost=0.00..1.05 rows=5 width=406)	Seq Scan озночает, что мы считали всю таблицу. Seq Scan это фулл скан.
	Filter: (id = 1)					Filter озночает, что мы считали по индексу

	//ANALYZE Выполняет запрос и предоставляет план выполнения запроса  
	EXPLAIN ANALYZE SELECT * FROM ticket;

2:	//В pg_catalog есть таблица, которая хранит статистику на каждую нашу таблицу.
	  Сама таблица называется pg_class и именно эту таблицу стоимостной оптимизатор берет для рассчета плана выполнения запроса.
	  Пример того как мы можем отобразить статистику таблце ticket:	
									SELECT 
									reltuples, --Количество строк в таблице
  								        relkind, -- говорит что это строки в таблице 
        								relpages --Показывает сколько страниц занимает наша таблица

									FROM pg_class WHERE relname = 'ticket'
==========================================================================================================================================================

3:	//В sql есть 2 оптимизатора 1-Синтаксический(уже не используется) 2-Стоимостной(Используется)
	   СУБД использует стоимостной оптимизатор для определения плана выполнения запроса
	  :Стоимостной оптимизатор рассчитывает стоимость операции на основании 2ух операций
	   // page_cost - как много страниц пришлость считать
	   // cpu_cost - как много операций над строками сделал наш процессор

	Рассчет cpu_cost осуществляется по формуле:  количество строк * 0,01 
	Рассчет page_cost осуществляется по формуле: количество страниц * 1.0 

	 SELECT
	 reltuples --Количество строк в таблице
  	 relkind, -- говорит что это строки в таблице 
         relpages --Показывает сколько страниц занимает наша таблица
	 FROM pg_class WHERE relname = 'ticket'


	(За основу взяли таблицу ticket где 55 записей)
	Расчет cpu_cost: reltuples * 0.01 = 0.55
	Расчте page_cost: relpages * 1.0 = 1
