Все команды мы должны вводить только тогда, когда мы находимя в директории maven проекта
mvn имяПлагины:help		// Покажет какие голы есть у плагина

mvn archetype:help		//Позволяет посмотреть все голы. у archetype 7 плагинов и один из нах называется archetype:generate.
mvn archetype:generate		//Создает новый проект на основе архетипа или обновляет фактический проект
mvn help:effective-pom		//Собирает(Склеиваем) superPOM и наш POM в один POM (именно с этим pom файлом мы будем работать в дальнейшем)
mvn dependency:analyze		//Анализирует все плагины нашего POM.xml, Показывает какие зависимости используются а какие нет(их стоит убрать)
mvn dependency:tree		//Визуализирует все наши зависимости ввиде дерева(покажет наши зависимости ввиде дерева)
mvn test -Dmaven.test.skip=true //Позволяет пропустить тесты такие как testCompile, testResources
